ion pollStatus(id) { const pollUrl = `${MODEL_URL}/${id}`; const check = setInterval(async () => { try { const response = await fetch(pollUrl, { headers: { 'Authorization': `Token ${API_TOKEN}` } }); const data = await response.json(); statusText.innerText = `Status: ${data.status}...`; if (data.status === "succeeded") { clearInterval(check); showVideo(data.output); // 'output' costuma ser a URL do vídeo ou array de URLs } else if (data.status === "failed") { clearInterval(check); throw new Error("A IA falhou em processar o vídeo."); } } catch (e) { clearInterval(check); alert(e.message); resetUI(); } }, 4000); // Verifica a cada 4 segundos } function showVideo(videoUrl) { loading.classList.add('hidden'); videoContainer.classList.remove('hidden'); // Algumas APIs retornam array, pegamos a primeira posição videoPlayer.src = Array.isArray(videoUrl) ? videoUrl[0] : videoUrl; videoPlayer.load(); } function resetUI() { generateBtn.disabled = false; generateBtn.classList.remove('opacity-50', 'cursor-not-allowed'); resultArea.classList.add('hidden'); }
